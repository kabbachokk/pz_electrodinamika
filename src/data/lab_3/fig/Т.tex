        declare function={ 
            psi(\x) = asin((sin(\x)*4.4^(1/2))/(1^(1/2))); 
        },
    ]
        \begin{axis}[
            grid,
            legend style={fill=white, fill opacity=0, draw opacity=0, text opacity=1},
            xlabel = {$\varphi$},
            domain = 0:28.475,
            restrict y to domain = -5:5,
            restrict x to domain = 0:28.475,
            xmin = 0,
            xmax = 90,
            ymin = 0,
            ymax = 4,
            samples = 2847,
            unbounded coords=jump
        ]
            \addlegendentry{$\lvert \textup{Т} (\varphi) \lvert$}
            \addplot[blue, thick] {
                (2*sin(psi(x))*cos(x))/sin(psi(x)+x)
            }
            node[pos=1,right,black]{$\perp$};

            \addplot[red, densely dashed, thick] {
                (2*sin(psi(x))*cos(x))/(sin(psi(x)+x)*cos(psi(x)-x))
            }
            node[pos=.85,left,black]{$\parallel$};

            \addplot[mark=none, black, samples=2, dashed] coordinates {(28.475,4) (28.475,0)}
            node[pos=.85,right,black]{$\varphi_\textup{кр}$};
        \end{axis}
    \end{tikzpicture}
